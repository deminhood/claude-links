// Load state
function loadState() {
try {
return JSON.parse(localStorage.getItem('fsroadmap') || '{}');
} catch { return {}; }
}

function saveState(state) {
try { localStorage.setItem('fsroadmap', JSON.stringify(state)); } catch {}
}

let state = loadState();

function totalSteps() {
return phases.reduce((acc, p) => acc + p.steps.length, 0);
}

function completedSteps() {
return Object.values(state).filter(Boolean).length;
}

function updateProgress() {
const total = totalSteps();
const done = completedSteps();
const pct = total ? (done / total * 100) : 0;
document.getElementById('progressBar').style.width = pct + '%';
document.getElementById('progressText').textContent = `${done} / ${total} complete`;
document.getElementById('doneMsg').classList.toggle('show', done === total && total > 0);
}

function getBadge(type) {
if (type === 'build') return '<span class="step-badge badge-build">BUILD</span>';
if (type === 'concept') return '<span class="step-badge badge-concept">CONCEPT</span>';
return '<span class="step-badge badge-tool">TOOL</span>';
}

function renderStep(step) {
const done = !!state[step.id];
const div = document.createElement('div');
div.className = 'step' + (done ? ' done' : '');
div.id = step.id;

div.innerHTML = `
<div class="step-top">
<div class="step-check">${done ? '✓' : ''}</div>
<div class="step-info">
<div class="step-title">${step.title}</div>
<div class="step-meta">
${getBadge(step.type)}
<span>${step.layer}</span>
<span>${step.time}</span>
</div>
</div>
<div class="step-expand-icon">▾</div>
</div>
<div class="step-detail">
<div class="step-detail-inner">
<div class="detail-section">
<div class="detail-label">Why this matters</div>
<div class="detail-text">${step.why}</div>
</div>
<div class="detail-section">
<div class="detail-label">What to build</div>
<div class="build-box">${step.build}</div>
</div>
${step.resources.length ? `
<div class="detail-section">
<div class="detail-label">Where to learn it</div>
<div class="resource-list">
${step.resources.map(r => `<a class="resource-link" href="${r.url}" target="_blank">↗ ${r.label}</a>`).join('')}
</div>
</div>` : ''}
<button class="mark-btn" data-id="${step.id}">${done ? 'Mark incomplete' : 'Mark complete ✓'}</button>
</div>
</div>
`;

// Toggle open/close
div.querySelector('.step-top').addEventListener('click', () => {
div.classList.toggle('open');
});

div.querySelector('.mark-btn').addEventListener('click', (e) => {
e.stopPropagation();
state[step.id] = !state[step.id];
saveState(state);
render();
});

return div;
}

function render() {
const container = document.getElementById('roadmap');
const openPhases = {};
const openSteps = {};

// Save current open states
container.querySelectorAll('.phase').forEach(el => {
openPhases[el.id] = !el.classList.contains('collapsed');
});
container.querySelectorAll('.step').forEach(el => {
openSteps[el.id] = el.classList.contains('open');
});

container.innerHTML = '';

phases.forEach(phase => {
const phaseEl = document.createElement('div');
phaseEl.className = 'phase';
phaseEl.id = phase.id;

const completedInPhase = phase.steps.filter(s => state[s.id]).length;
const isCollapsed = openPhases[phase.id] === false;
if (isCollapsed) phaseEl.classList.add('collapsed');

const bodyEl = document.createElement('div');
bodyEl.className = 'phase-body';

phase.steps.forEach(step => {
const stepEl = renderStep(step);
if (openSteps[step.id]) stepEl.classList.add('open');
bodyEl.appendChild(stepEl);
});

phaseEl.innerHTML = `
<div class="phase-header">
<span class="phase-num">${phase.label.split('—')[0].trim()}</span>
<span class="phase-label">${phase.label.split('—')[1]?.trim() || ''}</span>
<span class="phase-tag" style="color:${phase.tagColor};border-color:${phase.tagColor}40;background:${phase.tagColor}10">${completedInPhase}/${phase.steps.length}</span>
<span class="phase-chevron">▾</span>
</div>
<div class="phase-line"></div>
